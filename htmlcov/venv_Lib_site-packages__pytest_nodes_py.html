45def iterparentnodeids(nodeid: str) -> Iterator[str]: 

46 """Return the parent node IDs of a given node ID, inclusive. 

47 

48 For the node ID 

49 

50 "testing/code/test_excinfo.py::TestFormattedExcinfo::test_repr_source" 

51 

52 the result would be 

53 

54 "" 

55 "testing" 

56 "testing/code" 

57 "testing/code/test_excinfo.py" 

58 "testing/code/test_excinfo.py::TestFormattedExcinfo" 

59 "testing/code/test_excinfo.py::TestFormattedExcinfo::test_repr_source" 

60 

61 Note that :: parts are only considered at the last / component. 

62 """ 

63 pos = 0 

64 sep = SEP 

65 yield "" 

66 while True: 

67 at = nodeid.find(sep, pos) 

68 if at == -1 and sep == SEP: 

69 sep = "::" 

70 elif at == -1: 

71 if nodeid: 

72 yield nodeid 

73 break 

74 else: 

75 if at: 

76 yield nodeid[:at] 

77 pos = at + len(sep)
