112def getfuncargnames( 

113 function: Callable[..., Any], 

114 *, 

115 name: str = "", 

116 is_method: bool = False, 

117 cls: Optional[type] = None, 

118) -> Tuple[str, ...]: 

119 """Return the names of a function's mandatory arguments. 

120 

121 Should return the names of all function arguments that: 

122 * Aren't bound to an instance or type as in instance or class methods. 

123 * Don't have default values. 

124 * Aren't bound with functools.partial. 

125 * Aren't replaced with mocks. 

126 

127 The is_method and cls arguments indicate that the function should 

128 be treated as a bound method even though it's not unless, only in 

129 the case of cls, the function is a static method. 

130 

131 The name parameter should be the original name in which the function was collected. 

132 """ 

133 # TODO(RonnyPfannschmidt): This function should be refactored when we 

134 # revisit fixtures. The fixture mechanism should ask the node for 

135 # the fixture names, and not try to obtain directly from the 

136 # function object well after collection has occurred. 

137 

138 # The parameters attribute of a Signature object contains an 

139 # ordered mapping of parameter names to Parameter instances. This 

140 # creates a tuple of the names of the parameters that don't have 

141 # defaults. 

142 try: 

143 parameters = signature(function).parameters 

144 except (ValueError, TypeError) as e: 

145 fail( 

146 f"Could not determine arguments of {function!r}: {e}", pytrace=False, 

147 ) 

148 

149 arg_names = tuple( 

150 p.name 

151 for p in parameters.values() 

152 if ( 

153 p.kind is Parameter.POSITIONAL_OR_KEYWORD 

154 or p.kind is Parameter.KEYWORD_ONLY 

155 ) 

156 and p.default is Parameter.empty 

157 ) 

158 if not name: 

159 name = function.__name__ 

160 

161 # If this function should be treated as a bound method even though 

162 # it's passed as an unbound method or function, remove the first 

163 # parameter name. 

164 if is_method or ( 

165 cls and not isinstance(cls.__dict__.get(name, None), staticmethod) 

166 ): 

167 arg_names = arg_names[1:] 

168 # Remove any names that will be replaced with mocks. 

169 if hasattr(function, "__wrapped__"): 

170 arg_names = arg_names[num_mock_patch_args(function) :] 

171 return arg_names

216def ascii_escaped(val: Union[bytes, str]) -> str: 

217 r"""If val is pure ASCII, return it as an str, otherwise, escape 

218 bytes objects into a sequence of escaped bytes: 

219 

220 b'\xc3\xb4\xc5\xd6' -> r'\xc3\xb4\xc5\xd6' 

221 

222 and escapes unicode objects into a sequence of escaped unicode 

223 ids, e.g.: 

224 

225 r'4\nV\U00043efa\x0eMXWB\x1e\u3028\u15fd\xcd\U0007d944' 

226 

227 Note: 

228 The obvious "v.decode('unicode-escape')" will return 

229 valid UTF-8 unicode if it finds them in bytes, but we 

230 want to return escaped bytes for any byte, even if they match 

231 a UTF-8 string. 

232 """ 

233 if isinstance(val, bytes): 

234 ret = _bytes_to_ascii(val) 

235 else: 

236 ret = val.encode("unicode_escape").decode("ascii") 

237 return _translate_non_printable(ret)

252def get_real_func(obj):

254 functools.wraps or functools.partial.""" 

255 start_obj = obj 

256 for i in range(100): 

257 # __pytest_wrapped__ is set by @pytest.fixture when wrapping the fixture function

368# Perform exhaustiveness checking. 

369# 

370# Consider this example: 

371# 

372# MyUnion = Union[int, str] 

373# 

374# def handle(x: MyUnion) -> int { 

375# if isinstance(x, int): 

376# return 1 

377# elif isinstance(x, str): 

378# return 2 

379# else: 

380# raise Exception('unreachable') 

381# 

382# Now suppose we add a new variant: 

383# 

384# MyUnion = Union[int, str, bytes] 

385# 

386# After doing this, we must remember ourselves to go and update the handle 

387# function to handle the new variant. 

388# 

389# With `assert_never` we can do better: 

390# 

391# // raise Exception('unreachable') 

392# return assert_never(x) 

393# 

394# Now, if we forget to handle the new variant, the type-checker will emit a 

395# compile-time error, instead of the runtime error we would have gotten 

396# previously. 

397# 

398# This also work for Enums (if you use `is` to compare) and Literals. 

399def assert_never(value: "NoReturn") -> "NoReturn": 

400 assert False, "Unhandled value: {} ({})".format(value, type(value).__name__)
